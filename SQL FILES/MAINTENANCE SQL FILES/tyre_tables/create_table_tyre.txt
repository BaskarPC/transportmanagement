CREATE TABLE COMMON_TYRETRANSACTION(JOB_ID NUMBER(3) REFERENCES JOBTAB1(JOBID),TRANSACTION_DATE DATE,INCHARGE_NAME VARCHAR2(20),REBUILT_REPAIR VARCHAR2(20),COMPANY_NAME VARCHAR2(25),BILL_NUMBER VARCHAR2(10),AMOUNT NUMBER(10,2));

CREATE TABLE TYRE_TRANSACTION(JOB_ID NUMBER(3) REFERENCES JOBTAB1(JOBID),TYRENUMBER VARCHAR2(20),REBUILT_REPAIR VARCHAR2(20),REASON VARCHAR2(20),ST_IND VARCHAR2(1),PURPOSE VARCHAR2(20));
